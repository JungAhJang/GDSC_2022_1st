n, m = tuple(map(int,input().split()))

A = list(map(int,input().split()))

def calcul(A) :
    hap = 0
    global m
    if m % 2 != 0 :
        while m > 0 :
            hap += A[m-1]
            m = m -1
    elif m % 2 == 0 :
        while m > 0 :
            hap += A[m-1]
            m = m//2
    return hap

print(calcul(A))
